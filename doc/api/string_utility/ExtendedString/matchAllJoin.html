String matchAllJoin(RegExp regex, {String onNotFound = ''}) {
  final matchedString = regex
      .allMatches(this)
      .map<String>((v) => v.group(0))
      .join(',')
      .replaceAll(',', '');

  return matchedString.isEmpty ? onNotFound : matchedString;
}
